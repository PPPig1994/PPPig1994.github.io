<script type="text/javascript" src="jquery-3.1.1.min.js"></script>
<title>Document</title>
<script type="text/javascript">
    //$(function(){})相当于window.onload,代码写在{}之间
    $(function() {
        //1.选取button：$("button")
        //2.为button添加onclick响应函数：$("button").onclick(function(){})
        //代码写在function的{}中
        $("button").click(function() {

            alert("helloworld");
            alert(this.firstChild.nodeValue);
            alert($(this).text());
            //jquery和DOM对象
            //1.有jquery对象转为DOM对象
            //1).获取一个jquery对象
            var $btn = $("button");

            //2).jQuery对象是一个数组
            alert($btn.length);
            //3).可以通过数组的下标转为DOM对象
            alert($btn[0].firstChild.nodeValue);
            //2.由DOM对象转为jQuery对象
            //1).选取一个DOM对象
            var btn = document.getElementById("btn");
            //2).把DOM对象转为一个jQuery对象：使用$()进行包装
            alert("--" + $(btn).text());
        });
    })
</script>
</head>

<script src="jquery-3.1.1.min.js"></script>

        background-color: darkgray;

        font-size: 10px;
        float: left;

<script>
    $(function() {
        $("#btn1").click(function() {
            /* 选择body内的所有div元素 */
            $("body div").css("background", "#ffbbaa");

        });
        $("#btn2").click(function() {
            /* 在body内，选择子元素是div的 */
            $("body >div").css("background", "#ffbbaa");

        });

        $("#btn3").click(function() {
            /*  选择id为one的下一个div的元素 ,即兄弟*/
            $("#one+div").css("background", "#ffbbaa");

        });
        $("#btn4").click(function() {
            /* 选择id为two的元素后面的所有的div兄弟元素,即兄弟 */
            $("#two ~div").css("background", "#ffbbaa");

        });
        $("#btn5").click(function() {
            /* 选择id为two的元素所有div兄弟元素 */
            $("#two").siblings("div").css("background", "#ffbbaa");
        });
        $("#btn6").click(function() {
            /* 选择id为one的下一个span元素,以下选择器选择的是紧邻one的span元素，若该
            span和#one不相邻，选择器无效 */
            //$("#one +span").css("background", "#ffbbaa");
            $("#one").nextAll("span:first").css("background", "#ffbbaa");

        });
        $("#btn7").click(function() {
            /* 选择id为two的元素前边的所有div兄弟元素 */
            $("#two").prevAll("div").css("background", "#ffbbaa");
        });

    })
</script>
</head>

            $("#mover").slideToggle("slow", anmateIt);

            /* 选择第一个div元素 */
            $("div:first").css("background", "#ffbbaa");

        });
        $("#btn2").click(function() {
            /* 选择最后一个div元素 */
            $("div:last").css("background", "#ffbbaa");

        });

        $("#btn3").click(function() {
            /*  选择class不为one的所有元素*/
            $("div:not(.one)").css("background", "#ffbbaa");

        });
        $("#btn4").click(function() {
            /* 选择索引值为偶数的div元素 ，索引是从0开始的*/
            $("div:even").css("background", "#ffbbaa");

        });
        $("#btn5").click(function() {
            /* 选择索引值为奇数的div元素 */
            $("div:odd").css("background", "#ffbbaa");
        });
        $("#btn6").click(function() {
            /* 选择索引值大于3的div元素 */

            $("div:gt(3)").css("background", "#ffbbaa");

        });
        $("#btn7").click(function() {
            /* 选择索引值为等于3的div元素 */
            $("div:eq(3)").css("background", "#ffbbaa");
        });

        $("#btn8").click(function() {
            /* 选怎索引值小于3的div元素 */
            $("div:lt(3)").css("background", "#ffbbaa");
        });
        $("#btn9").click(function() {
            /* 选择所有的标题元素 */

            $(":header").css("background", "#ffbbaa");

        });
        $("#btn10").click(function() {
            /* 选择正在执行动画的所有元素 */
            $(":animated").css("background", "#ffbbaa");
        });
        $("#btn11").click(function() {
            /* 选择id为two的下一个span元素 */
            $("#two").nextAll("span:first").css("background", "#ffbbaa");
        });

    })
</script>

    }
</style>
<script>
    $(document).ready(function() {

        $("#btn1").click(function() {
            /* 选取含有属性title的div元素*/
            $("div[title]").css("background", "#ffbbaa");

        });
        $("#btn2").click(function() {
            /*选取属性title值等于test的div元素 */
            $("div[title='test']").css("background", "#ffbbaa");

        });

        $("#btn3").click(function() {
            /* 选取属性title值不等于test的div元素（没有属性title也将被选中）*/
            $("div[title!='test']").css("background", "#ffbbaa");

        });
        $("#btn4").click(function() {
            /* 选取属性title值以te开始的div元素*/

            $("div[title^='te']").css("background", "#ffbbaa");

        });
        $("#btn5").click(function() {
            /*选取属性title值以est结束的div元素 */
            $("div[title$='est']").css("background", "#ffbbaa");
        });
        $("#btn6").click(function() {
            /* 选取属性title值含有es的div元素 */

            $("div[title*='es']").css("background", "#ffbbaa");

        });
        $("#btn7").click(function() {
            /*组合属性选择器首先选区有属性id的div元素，然后在结果中选择属性title值含有es的div元素 */
            $("div[id][title*='es']").css("background", "#ffbbaa");
        });
        $("#btn8").click(function() {
            /*选取含有title属性值，且title属性值不等于test的div元素 */
            $("div[title][title!='test']").css("background", "#ffbbaa");
        });
    })
</script>

    $(function() {
        $("#btn1").click(function() {
            //对表单内可用input赋值操作，对可见单行文本框的值改变
            $(":text:enabled").val("文喵喵");
        });
        $("#btn2").click(function() {
            //对表单内不可用input赋值操作，对不可见单行文本框的值改变
            $(":text:disabled").val("文喵喵");
        });
        $("#btn3").click(function() {
            //获取多选框中的个数
            var num = $(":checkbox[name='newsletter']:checked").length;
            alert(num);
        });

        $("#btn4").click(function() {
            //获取多选框中的内容
            $(":checkbox[name='newsletter']:checked").each(function() {
                alert(this.value);

            });

        });


        $("#btn5").click(function() {
            //获取下拉框选中的内容
            /* 
            实际被选择的不是select，二十select的option子节点

            因为 $("select :selected")选择的是一个数组

            jQuery对象遍历的方式是使用each，在each内部的this是正在

            $("select :selected").each(function() {
